CREATE OR REPLACE PROCEDURE GET_GRADE
(
	SCORE NUMBER --외부에서 전달 받을 값
)
IS 
	GRADE VARCHAR2(5); -- BEGIN에서 쓸 값
BEGIN
	IF SCORE >=90 THEN GRADE :='A';
	ELSIF SCORE >=80 THEN GRADE :='B';
	ELSIF SCORE >=70 THEN GRADE :='C';
	ELSIF SCORE >=60 THEN GRADE :='D';
	ELSE GRADE :='F';
	END IF;

	DBMS_OUTPUT.ENABLE;
	DBMS_OUTPUT.PUT_LINE('당신의점수: '||SCORE||'점'||CHR(10)||'학점: '||GRADE);
END;

CALL GET_GRADE(100);

CALL MY_NEW_JOB_PROC('IT','DEVELOPER',5000,20000); 

CREATE OR REPLACE PROCEDURE DEL_JOB_PROC
(
	--TABLE에서 중복이없는게 PK값이니 PK값을 외부에서 받아올 생각을 해야한다.
	P_JOB_ID IN JOBS.JOB_ID%TYPE
)
IS
	CNT NUMBER :=0;
BEGIN
	SELECT COUNT(JOB_ID) INTO CNT FROM JOBS WHERE JOB_ID = P_JOB_ID;
	IF CNT!=0 THEN --삭제할게있다.
		DELETE FROM JOBS WHERE JOB_ID = P_JOB_ID;
		DBMS_OUTPUT.ENABLE;
		DBMS_OUTPUT.PUT_LINE('DELETE ALL DONE ABOUT'||' '||P_JOB_ID);
	ELSE
		DBMS_OUTPUT.ENABLE;
		DBMS_OUTPUT.PUT_LINE('NOT EXIST'||' '||P_JOB_ID);
	END IF;
END;

CALL DEL_JOB_PROC('IT');
